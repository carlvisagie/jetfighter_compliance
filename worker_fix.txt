from apscheduler.schedulers.background import BackgroundScheduler

scheduler = None

def start_worker():
    global scheduler

    if scheduler:
        return

    scheduler = BackgroundScheduler(timezone="UTC")

    scheduler.add_job(sweep_queue, "interval", seconds=10, id="queue")
    scheduler.add_job(check_slas, "interval", minutes=5, id="sla")

    # --- Auto-jobs (idempotent) ---
    try:
        scheduler.add_job(
            nightly_exports,
            "cron",
            hour=2,
            minute=0,
            id="nightly_exports"
        )
    except Exception:
        pass

    try:
        scheduler.add_job(
            weekly_digest,
            "cron",
            day_of_week="fri",
            hour=9,
            minute=0,
            id="weekly_digest"
        )
    except Exception:
        pass

    scheduler.start()
